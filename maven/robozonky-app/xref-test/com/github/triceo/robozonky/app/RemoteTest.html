142         final Configuration ctx = RemoteTest.mockConfiguration(false);

149         final Loan loan = Mockito.mock(Loan.class);

153         Mockito.when(zonky.getWallet()).thenReturn(Mockito.mock(Wallet.class));

159         Assertions.assertThat(result.get()).isEmpty();

162     @Test
163     public void strategyExecutionUnderBalance() {
164         // a lot of mocking to exercise the basic path all the way through to the core
165         final Loan l = Mockito.mock(Loan.class);
166         Mockito.when(l.getDatePublished()).thenReturn(OffsetDateTime.now());
167         Mockito.when(l.getRemainingInvestment()).thenReturn(1000.0);
168         final Configuration ctx = RemoteTest.mockConfiguration(true);
169         Mockito.when(ctx.getZonkyProxyBuilder()).thenReturn(new ZonkyProxy.Builder().asDryRun());
170         final SecretProvider secret = Mockito.mock(SecretProvider.class);
171         Mockito.when(secret.getPassword()).thenReturn("".toCharArray());
172         final AuthenticationHandler auth = AuthenticationHandler.passwordBased(secret);
173         final ZotifyApi cache = Mockito.mock(ZotifyApi.class);
174         Mockito.when(cache.getLoans()).thenReturn(Collections.singletonList(l));
175         final ApiProvider api = Mockito.mock(ApiProvider.class);
176         Mockito.when(api.cache()).thenReturn(cache);
177         Mockito.when(api.oauth()).thenReturn(Mockito.mock(ZonkyOAuthApi.class));
178         final Loan loan = Mockito.mock(Loan.class);
179         Mockito.when(loan.getDatePublished()).thenReturn(OffsetDateTime.now());
180         final ZonkyApi zonky = Mockito.mock(ZonkyApi.class);
181         Mockito.when(zonky.getLoan(ArgumentMatchers.anyInt())).thenReturn(loan);
182         final Wallet wallet = Mockito.mock(Wallet.class);
183         Mockito.when(wallet.getAvailableBalance()).thenReturn(BigDecimal.TEN);
184         Mockito.when(zonky.getWallet()).thenReturn(wallet);
185         Mockito.when(api.authenticated(ArgumentMatchers.any())).thenReturn(zonky);
186         // and now test
187         final Optional<Collection<Investment>> result = new Remote(ctx, auth).executeStrategy(api);
188         Assertions.assertThat(result).isPresent();
189         Assertions.assertThat(result.get()).isEmpty();
190         // check events
191         final List<Event> fired = Events.getFired();
192         final List<Event> lastFired = fired.subList(fired.size() - 5, fired.size());
193         final SoftAssertions softly = new SoftAssertions();
194         softly.assertThat(lastFired.get(0)).isInstanceOf(MarketplaceCheckStartedEvent.class);
195         softly.assertThat(lastFired.get(1)).isInstanceOf(LoanArrivedEvent.class);
196         softly.assertThat(lastFired.get(2)).isInstanceOf(MarketplaceCheckCompletedEvent.class);
197         softly.assertThat(lastFired.get(3)).isInstanceOf(ExecutionStartedEvent.class);
198         softly.assertThat(lastFired.get(4)).isInstanceOf(ExecutionCompletedEvent.class);
199         softly.assertAll();
200     }
