17  package com.github.triceo.robozonky.app;
18  
19  import java.util.Optional;
20  import java.util.concurrent.ExecutionException;
21  import java.util.concurrent.Future;
22  import java.util.function.Consumer;
23  
24  import com.github.triceo.robozonky.api.ReturnCode;
25  import com.github.triceo.robozonky.app.version.VersionIdentifier;
26  import org.assertj.core.api.Assertions;
27  import org.junit.Test;
28  import org.mockito.Mockito;
29  
30  public class VersionCheckerTest {
31  
32      @Test
33      public void checkShutdownHook() {
34          final VersionChecker v = new VersionChecker();
35          final Optional<Consumer<ReturnCode>> hook = v.get();
36          Assertions.assertThat(hook).isPresent();
37          hook.get().accept(ReturnCode.OK);
38      }
39  
40      @Test
41      public void versionCheckFailed() throws InterruptedException, ExecutionException {
42          final Future<VersionIdentifier> future = Mockito.mock(Future.class);
43          Mockito.doThrow(new InterruptedException()).when(future).get();
44          Assertions.assertThat(VersionChecker.newerRoboZonkyVersionExists(future)).isFalse();
45      }
46  
47      @Test
48      public void versionCheckSucceeded() throws InterruptedException, ExecutionException {
49          final VersionIdentifier version = Mockito.mock(VersionIdentifier.class);
50          Mockito.when(version.getLatestStable()).thenReturn("anything");
51          final Future<VersionIdentifier> future = Mockito.mock(Future.class);
52          Mockito.when(future.get()).thenReturn(version);
53          // current version is null, since it depends on JAR manifests and those are not available yet
54          Assertions.assertThat(VersionChecker.newerRoboZonkyVersionExists(future)).isFalse();
55      }
56  
57      @Test
58      public void versionCheckHasNewerStable() {
59          final String currentVersion = "1.0.0";
60          final String newerVersion = "1.0.1";
61          final VersionIdentifier v = Mockito.mock(VersionIdentifier.class);
62          Mockito.when(v.getLatestStable()).thenReturn(newerVersion);
63          Assertions.assertThat(VersionChecker.newerRoboZonkyVersionExists(v, currentVersion)).isTrue();
64      }
65  
66      @Test
67      public void versionCheckHasNewerUnstable() {
68          final String currentVersion = "1.0.0";
69          final String newerVersion = "1.0.1";
70          final VersionIdentifier v = Mockito.mock(VersionIdentifier.class);
71          Mockito.when(v.getLatestStable()).thenReturn(currentVersion);
72          Mockito.when(v.getLatestUnstable()).thenReturn(Optional.of(newerVersion));
73          Assertions.assertThat(VersionChecker.newerRoboZonkyVersionExists(v, currentVersion)).isTrue();
74      }
75  
76      @Test
77      public void versionCheckUpToDate() {
78          final String currentVersion = "1.0.0";
79          final VersionIdentifier v = Mockito.mock(VersionIdentifier.class);
80          Mockito.when(v.getLatestStable()).thenReturn(currentVersion);
81          Assertions.assertThat(VersionChecker.newerRoboZonkyVersionExists(v, currentVersion)).isFalse();
82      }
83  }
