43      private static final ZonkyApiToken TOKEN = new ZonkyApiToken(UUID.randomUUID().toString(),
44              UUID.randomUUID().toString(), 299);

46      private static SecretProvider getNewProvider() throws IOException, KeyStoreException {
47          final File file = File.createTempFile("robozonky-", ".keystore");
48          file.delete();
49          final String username = "user";
50          final char[] password = "pass".toCharArray();
51          final KeyStoreHandler ksh = KeyStoreHandler.create(file, password);
52          return SecretProvider.keyStoreBased(ksh, username, password);
53      }
54  
55      private static SecretProvider mockExistingProvider(final OffsetDateTime storedOn,
56                                                         final boolean succeedInSavingTokens,
57                                                         final boolean succeedInDeletingToken) throws JAXBException {
58          return AuthenticationHandlerTest.mockExistingProvider(ZonkyApiToken.marshal(AuthenticationHandlerTest.TOKEN),
59                  storedOn, succeedInSavingTokens, succeedInDeletingToken);
60      }
61  
62      private static SecretProvider mockExistingProvider(final String token, final OffsetDateTime storedOn,
63                                                         final boolean succeedInSavingTokens,
64                                                         final boolean succeedInDeletingToken) throws JAXBException {
65          final SecretProvider p = Mockito.mock(SecretProvider.class);
66          Mockito.when(p.getPassword()).thenReturn(new char[0]);
67          Mockito.when(p.getToken()).then(invocation -> Optional.of(new StringReader(token)));
68          Mockito.when(p.getTokenSetDate()).thenReturn(Optional.of(storedOn));
69          Mockito.when(p.setToken(ArgumentMatchers.any())).thenReturn(succeedInSavingTokens);
70          Mockito.when(p.deleteToken()).thenReturn(succeedInDeletingToken);
71          return p;
72      }

74      @Test
75      public void simplePasswordBased() throws IOException, KeyStoreException {
76          final SecretProvider secrets = AuthenticationHandlerTest.getNewProvider();
77          final AuthenticationHandler auth = AuthenticationHandler.passwordBased(secrets);
78          final ZonkyApi zonky = Mockito.mock(ZonkyApi.class);
79          final ZonkyOAuthApi zonkyOauth = Mockito.mock(ZonkyOAuthApi.class);
80          final ApiProvider apiProvider = Mockito.mock(ApiProvider.class);
81          Mockito.when(apiProvider.authenticated(ArgumentMatchers.any())).thenReturn(zonky);
82          Mockito.when(apiProvider.oauth()).thenReturn(zonkyOauth);
83          Assertions.assertThat(auth.getSecretProvider()).isSameAs(secrets);
84          Assertions.assertThat(auth.execute(apiProvider, (api) -> Collections.emptyList())).isPresent();
85          Mockito.verify(zonkyOauth, Mockito.times(1))
86                  .login(ArgumentMatchers.eq(secrets.getUsername()),
87                          ArgumentMatchers.eq(new String(secrets.getPassword())), ArgumentMatchers.any(),
88                          ArgumentMatchers.any());
89          Mockito.verify(zonkyOauth, Mockito.never())
90                  .refresh(ArgumentMatchers.any(), ArgumentMatchers.any(), ArgumentMatchers.any());
91          Mockito.verify(zonky, Mockito.times(1)).logout();
92      }
