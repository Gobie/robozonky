42  public class AuthenticationHandler {
43  
44      private static final Logger LOGGER = LoggerFactory.getLogger(AuthenticationHandler.class);
45  
46      /**
47       * Build authentication mechanism that will keep the session alive via the use of session token. The mechanism will
48       * never log out, but the session may expire if not refresh regularly. This is potentially unsafe, as it will
49       * eventually store a plain-text access token on the hard drive, for everyone to see.
50       *
51       * The token will only be refreshed if RoboZonky is launched between token expiration and X second before token
52       * expiration, where X comes from the arguments of this method.
53       *
54       * @param data Provider for the sensitive information, such as passwords and tokens.
55       * @param refreshAfter Access token will be refreshed after expiration minus this.
56       * @return This.
57       */
58      public static AuthenticationHandler tokenBased(final SecretProvider data, final TemporalAmount refreshAfter) {
59          return new AuthenticationHandler(data, true, refreshAfter);
60      }
61  
62      /**
63       * Build authentication mechanism that will log out at the end of RoboZonky's operations. This will ignore the
64       * access tokens.
65       *
66       * @param data Provider for the sensitive information, such as passwords and tokens.
67       * @return The desired authentication method.
68       */
69      public static AuthenticationHandler passwordBased(final SecretProvider data) {
70          return new AuthenticationHandler(data);
71      }
72  
73      private final boolean tokenBased;
74      private final SecretProvider data;
75      private final TemporalAmount tokenRefreshBeforeExpiration;
76  
77      private AuthenticationHandler(final SecretProvider data) {
78          this(data, false, Duration.ZERO);
79      }
80  
81      private AuthenticationHandler(final SecretProvider data, final boolean tokenBased,
82                                    final TemporalAmount tokenRefreshBeforeExpiration) {
83          this.data = data;
84          this.tokenRefreshBeforeExpiration = tokenRefreshBeforeExpiration;
85          this.tokenBased = tokenBased;
86      }
87  
88      public SecretProvider getSecretProvider() {
89          return data;
90      }
91  
92      private Authenticator buildWithPassword() {
93          if (!this.data.deleteToken()) { // get rid of any stale token
94              AuthenticationHandler.LOGGER.warn("Failed deleting token.");
95          }
96          return Authenticator.withCredentials(this.data.getUsername(), this.data.getPassword());
97      }
98  
99      /**
100      * Based on information received until this point, decide on the proper authentication method.
101      *
102      * @return Authentication method matching user preferences.
103      */
104     private Authenticator build() {
105         if (!this.tokenBased) {
106             AuthenticationHandler.LOGGER.debug("Password-based authentication requested.");
107             return this.buildWithPassword();
108         }
109         return this.data.getToken().map(r -> {
110             try {
111                 final ZonkyApiToken token = ZonkyApiToken.unmarshal(r);
112                 final OffsetDateTime obtained = this.data.getTokenSetDate()
113                         .orElse(OffsetDateTime.ofInstant(Instant.EPOCH, Defaults.ZONE_ID));
114                 final OffsetDateTime expires = obtained.plus(token.getExpiresIn(), ChronoUnit.SECONDS);
115                 AuthenticationHandler.LOGGER.debug("Token obtained on {}, expires on {}.", obtained, expires);
116                 final OffsetDateTime now = OffsetDateTime.now();
117                 if (expires.isBefore(now)) {
118                     AuthenticationHandler.LOGGER.debug("Token expired, using password-based authentication.");
119                     return this.buildWithPassword();
120                 } else if (expires.minus(this.tokenRefreshBeforeExpiration).isBefore(now)) {
121                     AuthenticationHandler.LOGGER.debug("Access token expiring, will be refreshed.");
122                     return Authenticator.withAccessTokenAndRefresh(this.data.getUsername(), token);
123                 } else {
124                     AuthenticationHandler.LOGGER.debug("Reusing access token.");
125                     return Authenticator.withAccessToken(this.data.getUsername(), token);
126                 }
127             } catch (final Exception ex) {
128                 AuthenticationHandler.LOGGER.warn("Failed parsing token, using password-based authentication.", ex);
129                 return this.buildWithPassword();
130             }
131         }).orElseGet(() -> {  // no token available, also using password-based
132             AuthenticationHandler.LOGGER.debug("Token not available, using password-based authentication.");
133             return this.buildWithPassword();
134         });
135     }
136 
137     boolean storeToken(final ZonkyApiToken token) throws JAXBException {
138         final String marshalled = ZonkyApiToken.marshal(token);
139         final boolean tokenStored = this.data.setToken(new StringReader(marshalled));
140         if (!tokenStored) {
141             AuthenticationHandler.LOGGER.debug("Failed storing token.");
142         }
143         return tokenStored;
144     }
145 
146     /**
147      * Decide whether or not to log out, based on user preferences.
148      *
149      * @param token Token in question.
150      * @return True if RoboZonky should log out, false otherwise.
151      */
152     private boolean isLogoutAllowed(final ZonkyApiToken token) {
153         if (!this.tokenBased) { // not using token; always logout
154             return true;
155         }
156         final Optional<Reader> tokenStream = this.data.getToken();
157         if (tokenStream.isPresent()) { // token already exists, do not logout
158             return false;
159         } else try {
160             return !this.storeToken(token);
161         } catch (final Exception ex) {
162             AuthenticationHandler.LOGGER.info("Failed writing access token, will need to use password next time.", ex);
163             return true;
164         }
165     }
166 
167     public Optional<Collection<Investment>> execute(final ApiProvider provider,
168                                                     final Function<ZonkyApi, Collection<Investment>> operation) {
169         final Authentication login;
170         try { // catch this exception here, so that anything coming from the invest() method can be thrown separately
171             login = this.build().authenticate(provider);
172         } catch (final BadRequestException ex) {
173             AuthenticationHandler.LOGGER.error("Failed authenticating with Zonky.", ex);
174             return Optional.empty();
175         }
176         final Collection<Investment> result = operation.apply(login.getZonkyApi());
177         try { // log out and ignore any resulting error
178             final boolean logoutAllowed = this.isLogoutAllowed(login.getZonkyApiToken());
179             if (logoutAllowed) {
180                 login.getZonkyApi().logout();
181             } else { // if we're using the token, we should never log out
182                 AuthenticationHandler.LOGGER.info("Refresh token needs to be reused, not logging out of Zonky.");
183             }
184         } catch (final RuntimeException ex) {
185             AuthenticationHandler.LOGGER.warn("Failed logging out of Zonky.", ex);
186         }
187         return Optional.of(result);
188     }
