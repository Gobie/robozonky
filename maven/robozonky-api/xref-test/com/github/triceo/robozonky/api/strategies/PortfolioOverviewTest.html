17  package com.github.triceo.robozonky.api.strategies;
18  
19  import java.math.BigDecimal;
20  import java.math.RoundingMode;
21  import java.util.Arrays;
22  import java.util.Collections;
23  import java.util.List;
24  
25  import com.github.triceo.robozonky.api.remote.entities.Investment;
26  import com.github.triceo.robozonky.api.remote.entities.RiskPortfolio;
27  import com.github.triceo.robozonky.api.remote.entities.Statistics;
28  import com.github.triceo.robozonky.api.remote.enums.Rating;
29  import org.assertj.core.api.Assertions;
30  import org.assertj.core.api.SoftAssertions;
31  import org.junit.Test;
32  import org.mockito.Mockito;
33  
34  public class PortfolioOverviewTest {
35  
36      private static void assertProperRatingShare(final PortfolioOverview result, final Rating r, final int amount,
37                                                  final int total) {
38          final BigDecimal expectedShare =
39                  BigDecimal.valueOf(amount).divide(BigDecimal.valueOf(total), 4, RoundingMode.HALF_EVEN);
40          Assertions.assertThat(result.getShareOnInvestment(r)).isEqualTo(expectedShare);
41      }
42  
43      private static List<Investment> getMockInvestmentWithBalance(final int loanAmount) {
44          final Investment i = Mockito.mock(Investment.class);
45          Mockito.when(i.getAmount()).thenReturn(loanAmount);
46          return Collections.singletonList(i);
47      }
48  
49      @Test
50      public void emptyPortfolio() {
51          final Statistics s = Mockito.mock(Statistics.class);
52          final PortfolioOverview o = PortfolioOverview.calculate(BigDecimal.valueOf(5000.0), s, Collections.emptyList());
53          final SoftAssertions softly = new SoftAssertions();
54          for (final Rating r: Rating.values()) {
55              softly.assertThat(o.getShareOnInvestment(r)).isEqualTo(BigDecimal.ZERO);
56          }
57      }
58  
59      @Test
60      public void properRatingShareCalculation() {
61          // mock necessary structures
62          final int amountAA = 300, amountB = 200, amountD = 100;
63          final int totalPie = amountAA + amountB + amountD;
64          final RiskPortfolio riskAA = new RiskPortfolio(Rating.AA, 0, amountAA, 0);
65          final RiskPortfolio riskB = new RiskPortfolio(Rating.B, 0, amountB, 0);
66          final RiskPortfolio riskD = new RiskPortfolio(Rating.D, 0, amountD, 0);
67          final Statistics stats = Mockito.mock(Statistics.class);
68          Mockito.when(stats.getRiskPortfolio()).thenReturn(Arrays.asList(riskAA, riskB, riskD));
69  
70          // check standard operation
71          final BigDecimal balance = BigDecimal.TEN;
72          PortfolioOverview result = PortfolioOverview.calculate(balance, stats, Collections.emptyList());
73          Assertions.assertThat(result.getCzkAvailable()).isEqualTo(balance.intValue());
74          PortfolioOverviewTest.assertProperRatingShare(result, Rating.AA, amountAA, totalPie);
75          PortfolioOverviewTest.assertProperRatingShare(result, Rating.B, amountB, totalPie);
76          PortfolioOverviewTest.assertProperRatingShare(result, Rating.D, amountD, totalPie);
77          PortfolioOverviewTest.assertProperRatingShare(result, Rating.AAAAA, 0, totalPie); // test other ratings included
78          PortfolioOverviewTest.assertProperRatingShare(result, Rating.AAAA, 0, totalPie);
79          PortfolioOverviewTest.assertProperRatingShare(result, Rating.AAA, 0, totalPie);
80          PortfolioOverviewTest.assertProperRatingShare(result, Rating.A, 0, totalPie);
81          PortfolioOverviewTest.assertProperRatingShare(result, Rating.C, 0, totalPie);
82          // check operation with offline investments
83          final int increment = 200, newTotalPie = totalPie + increment;
84          final List<Investment> investments = PortfolioOverviewTest.getMockInvestmentWithBalance(increment);
85          final Investment i = investments.get(0);
86          Mockito.when(i.getRating()).thenReturn(Rating.A);
87          result = PortfolioOverview.calculate(balance, stats, investments);
88          Assertions.assertThat(result.getCzkAvailable()).isEqualTo(balance.intValue());
89          PortfolioOverviewTest.assertProperRatingShare(result, Rating.AA, amountAA, newTotalPie);
90          PortfolioOverviewTest.assertProperRatingShare(result, Rating.B, amountB, newTotalPie);
91          PortfolioOverviewTest.assertProperRatingShare(result, Rating.D, amountD, newTotalPie);
92          PortfolioOverviewTest.assertProperRatingShare(result, Rating.AAAAA, 0, newTotalPie); // test other ratings included
93          PortfolioOverviewTest.assertProperRatingShare(result, Rating.AAAA, 0, newTotalPie);
94          PortfolioOverviewTest.assertProperRatingShare(result, Rating.AAA, 0, newTotalPie);
95          PortfolioOverviewTest.assertProperRatingShare(result, Rating.A, increment, newTotalPie);
96          PortfolioOverviewTest.assertProperRatingShare(result, Rating.C, 0, newTotalPie);
97      }
