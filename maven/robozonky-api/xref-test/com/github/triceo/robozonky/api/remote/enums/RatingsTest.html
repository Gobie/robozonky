17  package com.github.triceo.robozonky.api.remote.enums;
18  
19  import org.assertj.core.api.Assertions;
20  import org.assertj.core.api.SoftAssertions;
21  import org.junit.Test;
22  
23  public class RatingsTest {
24  
25      @Test
26      public void ratingsAreProperlySerialized() {
27          Assertions.assertThat(Ratings.of(Rating.A).toString()).isEqualTo("[\"A\"]");
28          Assertions.assertThat(Ratings.of(Rating.AA, Rating.AAAA, Rating.AA).toString()).isEqualTo("[\"AAAA\", \"AA\"]");
29      }
30  
31      @Test
32      public void ratingsAreProperlyEnumerated() {
33          Assertions.assertThat(Ratings.of(Rating.AAA).getRatings()).containsExactly(Rating.AAA);
34          Assertions.assertThat(Ratings.of(Rating.B, Rating.C, Rating.B).getRatings()).containsExactly(Rating.B, Rating.C);
35      }
36  
37      @Test
38      public void ofAll() {
39          final Ratings result = Ratings.all();
40          Assertions.assertThat(result.getRatings()).containsOnly(Rating.values());
41      }
42  
43      @Test
44      public void correctValueOf() {
45          // test no items
46          Assertions.assertThat(Ratings.valueOf("[]").getRatings()).isEmpty();
47          Assertions.assertThat(Ratings.valueOf(" [ ] ").getRatings()).isEmpty();
48          // test one item
49          Assertions.assertThat(Ratings.valueOf("[ \"A\" ]").getRatings())
50                  .containsExactly(Rating.A);
51          Assertions.assertThat(Ratings.valueOf(" [ \"AA\"]").getRatings())
52                  .containsExactly(Rating.AA);
53          // test multiple items
54          Assertions.assertThat(Ratings.valueOf(" [\"AAA\", \"B\"]").getRatings())
55                  .containsExactly(Rating.AAA, Rating.B);
56          Assertions.assertThat(Ratings.valueOf(" [\"B\" , \"AAA\"] ").getRatings())
57                  .containsExactly(Rating.AAA, Rating.B);
58      }
59  
60      @Test(expected = IllegalArgumentException.class)
61      public void invalidValueOf() {
62          Ratings.valueOf("[");
63      }
64  
65      @Test(expected = IllegalArgumentException.class)
66      public void unquotedValueOf() {
67          Ratings.valueOf("[A]");
68      }
69  
70      @Test(expected = IllegalArgumentException.class)
71      public void unknownValueOf() {
72          Ratings.valueOf("[\"SOME_UNKNOWN_VALUE\"]");
73      }
74  
75      @Test
76      public void equality() {
77          final Ratings r1 = Ratings.of(Rating.A, Rating.B);
78          Assertions.assertThat(r1).isEqualTo(r1);
79          final Ratings r2 = Ratings.of(Rating.A, Rating.B);
80          final SoftAssertions softly = new SoftAssertions();
81          softly.assertThat(r1).isNotSameAs(r2).isEqualTo(r2);
82          softly.assertThat(r2).isEqualTo(r1);
83          softly.assertAll();
84      }
85  }
