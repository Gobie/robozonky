17  package com.github.triceo.robozonky.api.strategies;
18  
19  import java.time.Duration;
20  import java.time.OffsetDateTime;
21  import java.time.temporal.TemporalAmount;
22  import java.util.Objects;
23  import java.util.Optional;
24  
25  import com.github.triceo.robozonky.api.Defaults;
26  import com.github.triceo.robozonky.api.confirmations.ConfirmationProvider;
27  import com.github.triceo.robozonky.api.remote.entities.Loan;
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  
31  /**
32   * Carries metadata regarding a {@link Loan}.
33   */
34  public final class LoanDescriptor {
35  
36      private static final Logger LOGGER = LoggerFactory.getLogger(LoanDescriptor.class);
37  
38      private final Loan loan;
39      private final OffsetDateTime captchaExpirationDateTime;
40  
41      public LoanDescriptor(final Loan loan, final TemporalAmount captchaProtectionDuration) {
42          this.loan = loan;
43          this.captchaExpirationDateTime = loan.getDatePublished().plus(captchaProtectionDuration);
44      }
45  
46      public LoanDescriptor(final Loan loan) {
47          this(loan, Duration.ZERO);
48      }
49  
50      public Loan getLoan() {
51          return this.loan;
52      }
53  
54      /**
55       * If protected by CAPTCHA, gives the first instant when the CAPTCHA protection is over.
56       *
57       * @return Present if loan protected by CAPTCHA (= currently always), otherwise empty.
58       */
59      public Optional<OffsetDateTime> getLoanCaptchaProtectionEndDateTime() {
60          return Optional.of(captchaExpirationDateTime);
61      }
62  
63      /**
64       * Convert the descriptor into an actual investment recommendation. This will be executed by the
65       * {@link InvestmentStrategy}.
66       * @param amount The amount recommended to invest.
67       * @param confirmationRequired Whether or not {@link ConfirmationProvider} is required to confirm the investment.
68       * @return Empty if amount is out of bounds.
69       */
70      public Optional<Recommendation> recommend(final int amount, final boolean confirmationRequired) {
71          if (amount >= Defaults.MINIMUM_INVESTMENT_IN_CZK && amount <= loan.getRemainingInvestment()) {
72              return Optional.of(new Recommendation(this, amount, confirmationRequired));
73          } else {
74              LoanDescriptor.LOGGER.warn("Can not recommend {} CZK with {} CZK remaining in loan #{}.", amount,
75                      loan.getRemainingInvestment(), loan.getId());
76              return Optional.empty();
77          }
78      }
79  
80      /**
81       * Convert the descriptor into an actual investment recommendation. This will be executed by the
82       * {@link InvestmentStrategy}.
83       * @param amount The amount recommended to invest.
84       * @return The value returned by {@link #recommend(int, boolean)}, where the last argument is false.
85       */
86      public Optional<Recommendation> recommend(final int amount) {
87          return recommend(amount, false);
88      }
89  
90      @Override
91      public String toString() {
92          return "LoanDescriptorImpl{" + "loan=" + loan +
93                  ", captchaExpirationDateTime=" + captchaExpirationDateTime +
94                  '}';
95      }
96  
97      @Override
98      public boolean equals(final Object o) {
99          if (this == o) {
100             return true;
101         }
102         if (!(o instanceof LoanDescriptor)) {
103             return false;
104         }
105         final LoanDescriptor that = (LoanDescriptor) o;
106         return Objects.equals(loan, that.loan);
107     }
108 
109     @Override
110     public int hashCode() {
111         return Objects.hash(loan);
112     }
113 
114 }
