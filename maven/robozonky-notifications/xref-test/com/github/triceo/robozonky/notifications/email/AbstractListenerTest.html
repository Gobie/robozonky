44  @RunWith(Parameterized.class)
45  public abstract class AbstractListenerTest {
46  
47      private static NotificationProperties getNotificationProperties() {
48          System.setProperty(NotificationProperties.CONFIG_FILE_LOCATION_PROPERTY,
49                  NotificationPropertiesTest.class.getResource("notifications-enabled.cfg").toString());
50          final Optional<NotificationProperties> p = NotificationProperties.getProperties();
51          System.clearProperty(NotificationProperties.CONFIG_FILE_LOCATION_PROPERTY);
52          return p.get();
53      }
54  
55      @Parameterized.Parameters(name = "{0}")
56      public static Collection<Object[]> getListeners() {
57          // prepare data
58          final Loan loan = Mockito.mock(Loan.class);
59          Mockito.when(loan.getId()).thenReturn(66666);
60          Mockito.when(loan.getDatePublished()).thenReturn(OffsetDateTime.now());
61          Mockito.when(loan.getRemainingInvestment()).thenReturn(2000.0);
62          Mockito.when(loan.getAmount()).thenReturn(100000.0);
63          final LoanDescriptor loanDescriptor = new LoanDescriptor(loan);
64          final Recommendation recommendation = loanDescriptor.recommend(1200, false).get();
65          final Investment i = new Investment(loan, 1000);
66          final NotificationProperties properties = AbstractListenerTest.getNotificationProperties();
67          // create events for listeners
68          final Map<Class<? extends Event>, Event> events = new HashMap<>(SupportedListener.values().length);
69          events.put(SupportedListener.INVESTMENT_DELEGATED.getEventType(),
70                  new InvestmentDelegatedEvent(recommendation, 200, "random"));
71          events.put(SupportedListener.INVESTMENT_MADE.getEventType(), new InvestmentMadeEvent(i, 200));
72          events.put(SupportedListener.INVESTMENT_REJECTED.getEventType(),
73                  new InvestmentRejectedEvent(recommendation, 200, "random"));
74          events.put(SupportedListener.EXECUTION_STARTED.getEventType(),
75                  new ExecutionStartedEvent(Collections.emptyList(), 200));
76          events.put(SupportedListener.CRASHED.getEventType(),
77                  new RoboZonkyCrashedEvent(ReturnCode.ERROR_UNEXPECTED, new RuntimeException()));
78          // create the listeners
79          return Stream.of(SupportedListener.values())
80                  .map(s -> new Object[] {s, s.getListener(properties), events.get(s.getEventType())})
81                  .collect(Collectors.toList());
82      }
83

85      public EventListener<Event> listener;
86      @Parameterized.Parameter(2)
87      public Event event;
88      // only exists so that the parameter can have a nice constant description. otherwise PIT will report 0 coverage.
89      @Parameterized.Parameter
90      public SupportedListener listenerType;
