39  /**
40   * There are the following kinds of proxies:
41   * <ul>
42   *     <li>DRY: One in a dry run.</li>
43   *     <li>SIMPLE: One that has no way of obtaining external investing input.</li>
44   *     <li>CONFIRMING: One that knows to obtain external investing input.</li>
45   * </ul>
46   *
47   * There are following kinds of investments:
48   * <ul>
49   *     <li>CONFIRMED: Require confirmation, and therefore requiring a CONFIRMING proxy. In case of
50   *     positive confirmation, we can invest locally.</li>
51   *     <li>UNCONFIRMED: Not requiring confirmation and not CAPTCHA-protected, these can be invested locally.</li>
52   * </ul>
53   *
54   * Further, investments can be protected by CAPTCHA:
55   * <ul>
56   *     <li>PROTECTED: Can not invest locally, require delegation and therefore a CONFIRMING proxy.</li>
57   *     <li>UNPROTECTED: Can invest locally, SIMPLE proxy will be enough provided that investment is not CONFIRMED.</li>
58   * </ul>
59   *
60   * Based on that, these are the possible states that need to be tested:
61   * <ul>
62   *     <li>DRY proxy, any investment, regardless of captcha. Always accepted</li>
63   *     <li>SIMPLE proxy:
64   *         <ul>
65   *             <li>Rejects PROTECTED investments since we cannot bypass CAPTCHA.</li>
66   *             <li>Fails on CONFIRMED investments since we cannot reach confirmation.</li>
67   *             <li>Fails on network issues.</li>
68   *             <li>Otherwise invests locally.</li>
69   *         </ul>
70   *     </li>
71   *     <li>CONFIRMING proxy:
72   *         <ul>
73   *             <li>Delegates PROTECTED investments.</li>
74   *             <li>Invests UNPROTECTED CONFIRMED investments locally or rejects them.</li>
75   *             <li>Invests UNPROTECTED UNCONFIRMED investments locally.</li>
76   *             <li>Fails on network issues.</li>
77   *         </ul>
78   *     </li>
79   * </ul>
80   *
81   * This test aims to test all of these various states.
82   */
83  @RunWith(Parameterized.class)
84  public class ZonkyProxyTest extends AbstractInvestingTest {

107     @Parameterized.Parameters(name = "{0}+{1}+{2}({3})={4}")
108     public static Collection<Object[]> generatePossibilities() {
109         final Collection<Object[]> result = new ArrayList<>();
110         // dry proxy
111         for (final ZonkyProxyTest.Captcha c1: ZonkyProxyTest.Captcha.values()) {
112             for (final ZonkyProxyTest.Remote c2: ZonkyProxyTest.Remote.values()) {
113                 result.add(new Object[] {ZonkyProxyTest.ProxyType.DRY, c1, c2, null, ZonkyResponseType.INVESTED});
114             }
115         }
116         // simple proxy
117         result.add(new Object[] {ZonkyProxyTest.ProxyType.SIMPLE, ZonkyProxyTest.Captcha.PROTECTED,
118                 ZonkyProxyTest.Remote.CONFIRMED, null, ZonkyResponseType.FAILED});
119         result.add(new Object[] {ZonkyProxyTest.ProxyType.SIMPLE, ZonkyProxyTest.Captcha.PROTECTED,
120                 ZonkyProxyTest.Remote.UNCONFIRMED, null, ZonkyResponseType.REJECTED});
121         result.add(new Object[] {ZonkyProxyTest.ProxyType.SIMPLE, ZonkyProxyTest.Captcha.UNPROTECTED,
122                 ZonkyProxyTest.Remote.CONFIRMED, null, ZonkyResponseType.FAILED});
123         result.add(new Object[] {ZonkyProxyTest.ProxyType.SIMPLE, ZonkyProxyTest.Captcha.UNPROTECTED,
124                 ZonkyProxyTest.Remote.UNCONFIRMED, null, ZonkyResponseType.INVESTED});
125         // confirming proxy, response present
126         result.add(new Object[] {ZonkyProxyTest.ProxyType.CONFIRMING, ZonkyProxyTest.Captcha.PROTECTED,
127                 ZonkyProxyTest.Remote.CONFIRMED, ZonkyProxyTest.RemoteResponse.PRESENT, ZonkyResponseType.DELEGATED});
128         result.add(new Object[] {ZonkyProxyTest.ProxyType.CONFIRMING, ZonkyProxyTest.Captcha.PROTECTED,
129                 ZonkyProxyTest.Remote.UNCONFIRMED, ZonkyProxyTest.RemoteResponse.PRESENT, ZonkyResponseType.DELEGATED});
130         result.add(new Object[] {ZonkyProxyTest.ProxyType.CONFIRMING, ZonkyProxyTest.Captcha.UNPROTECTED,
131                 ZonkyProxyTest.Remote.CONFIRMED, ZonkyProxyTest.RemoteResponse.PRESENT, ZonkyResponseType.INVESTED});
132         result.add(new Object[] {ZonkyProxyTest.ProxyType.CONFIRMING, ZonkyProxyTest.Captcha.UNPROTECTED,
133                 ZonkyProxyTest.Remote.UNCONFIRMED, ZonkyProxyTest.RemoteResponse.PRESENT, ZonkyResponseType.INVESTED});
134         // confirming proxy, network failure
135         result.add(new Object[] {ZonkyProxyTest.ProxyType.CONFIRMING, ZonkyProxyTest.Captcha.PROTECTED,
136                 ZonkyProxyTest.Remote.CONFIRMED, ZonkyProxyTest.RemoteResponse.ABSENT, ZonkyResponseType.FAILED});
137         result.add(new Object[] {ZonkyProxyTest.ProxyType.CONFIRMING, ZonkyProxyTest.Captcha.PROTECTED,
138                 ZonkyProxyTest.Remote.UNCONFIRMED, ZonkyProxyTest.RemoteResponse.ABSENT, ZonkyResponseType.FAILED});
139         result.add(new Object[] {ZonkyProxyTest.ProxyType.CONFIRMING, ZonkyProxyTest.Captcha.UNPROTECTED,
140                 ZonkyProxyTest.Remote.CONFIRMED, ZonkyProxyTest.RemoteResponse.ABSENT, ZonkyResponseType.FAILED});
141         result.add(new Object[] {ZonkyProxyTest.ProxyType.CONFIRMING, ZonkyProxyTest.Captcha.UNPROTECTED,
142                 ZonkyProxyTest.Remote.UNCONFIRMED, ZonkyProxyTest.RemoteResponse.ABSENT, ZonkyResponseType.INVESTED});
143         return Collections.unmodifiableCollection(result);
144     }

146     private static final double LOAN_AMOUNT = 2000.0;
147     private static final int CONFIRMED_AMOUNT = (int) (LOAN_AMOUNT / 2);
148 
149     private static LoanDescriptor mockLoanDescriptor(final boolean protectByCaptcha) {
150         final Loan loan = AbstractInvestingTest.mockLoan();
151         return new LoanDescriptor(loan, protectByCaptcha ? Duration.ofHours(1) : Duration.ZERO);
152     }

154     @Parameterized.Parameter
155     public ZonkyProxyTest.ProxyType proxyType;
156     @Parameterized.Parameter(1)
157     public ZonkyProxyTest.Captcha captcha;
158     @Parameterized.Parameter(2)
159     public ZonkyProxyTest.Remote confirmation;
160     @Parameterized.Parameter(3)
161     public ZonkyProxyTest.RemoteResponse confirmationResponse;
162     @Parameterized.Parameter(4)
163     public ZonkyResponseType responseType;
164 
165     private Recommendation getRecommendation() {
166         final LoanDescriptor ld =
167                 ZonkyProxyTest.mockLoanDescriptor(captcha == ZonkyProxyTest.Captcha.PROTECTED);
168         return ld.recommend(ZonkyProxyTest.CONFIRMED_AMOUNT,
169                 confirmation == ZonkyProxyTest.Remote.CONFIRMED).get();
170     }
171 
172     private ZonkyProxy getZonkyProxy() {
173         final ZonkyApi api = Mockito.mock(ZonkyApi.class);
174         switch (proxyType) {
175             case DRY:
176                 return new ZonkyProxy.Builder().asDryRun().build(api);
177             case SIMPLE:
178                 return new ZonkyProxy.Builder().build(api);
179             case CONFIRMING:
180                 final ConfirmationProvider cp = Mockito.mock(ConfirmationProvider.class);
181                 switch (confirmationResponse) {
182                     case PRESENT:
183                         final Confirmation c = (captcha == ZonkyProxyTest.Captcha.PROTECTED) ?
184                                 new Confirmation(ConfirmationType.DELEGATED) :
185                                 new Confirmation(ZonkyProxyTest.CONFIRMED_AMOUNT);
186                         Mockito.when(cp.requestConfirmation(ArgumentMatchers.any(), ArgumentMatchers.anyInt(),
187                                 ArgumentMatchers.anyInt())).thenReturn(Optional.of(c));
188                         break;
189                     case ABSENT:
190                         Mockito.when(cp.requestConfirmation(ArgumentMatchers.any(), ArgumentMatchers.anyInt(),
191                                 ArgumentMatchers.anyInt())).thenReturn(Optional.empty());
192                         break;
193                     default:
194                             throw new IllegalStateException();
195                 }
196                 return new ZonkyProxy.Builder().usingConfirmation(cp, "user").build(api);
197             default:
198                 throw new IllegalStateException();
199         }
200     }
201 
202     @Test
203     public void test() {
204         final Recommendation r = this.getRecommendation();
205         final ZonkyProxy p = this.getZonkyProxy();
206         final ZonkyResponse result = p.invest(r);
207         if (this.proxyType == ZonkyProxyTest.ProxyType.CONFIRMING) {
208             Assertions.assertThat(p.getConfirmationProvider()).isNotNull();
209         } else {
210             Assertions.assertThat(p.getConfirmationProvider()).isNull();
211         }
212         Assertions.assertThat(result.getType()).isEqualTo(responseType);
213         if (result.getType() == ZonkyResponseType.INVESTED) {
214             Assertions.assertThat(result.getConfirmedAmount()).hasValue(ZonkyProxyTest.CONFIRMED_AMOUNT);
215         } else {
216             Assertions.assertThat(result.getConfirmedAmount()).isEmpty();
217         }
218     }
