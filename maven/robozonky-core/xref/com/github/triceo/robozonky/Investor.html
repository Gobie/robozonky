57      /**
58       * The core investing call. Receives a particular loan, checks if the user has enough money to invest, and sends the
59       * command to the Zonky API.
60       *
61       * @param recommendation Recommendation to invest.
62       * @param api API to invest through.
63       * @param tracker Status of the investing session.
64       * @return Present if input valid and operation succeeded, empty otherwise.
65       */
66      static Optional<Investment> actuallyInvest(final Recommendation recommendation, final ZonkyProxy api,
67                                                 final InvestmentTracker tracker) {
68          final int amount = recommendation.getRecommendedInvestmentAmount();
69          final int loanId = recommendation.getLoanDescriptor().getLoan().getId();
70          final BigDecimal balance = tracker.getCurrentBalance();
71          if (balance.compareTo(BigDecimal.valueOf(amount)) < 0) { // strategy should not allow this
72              Investor.LOGGER.info("Not investing into loan #{}, {} CZK to invest is more than {} CZK balance.", loanId,
73                      amount, balance);
74              return Optional.empty();
75          }
76          Events.fire(new InvestmentRequestedEvent(recommendation));
77          final ZonkyResponse response = api.invest(recommendation);
78          switch (response.getType()) {
79              case REJECTED:
80                  Events.fire(new InvestmentRejectedEvent(recommendation, balance.intValue(),
81                          api.getConfirmationProvider().getId()));
82                  tracker.ignoreLoanForever(loanId);
83                  return Optional.empty();
84              case DELEGATED:
85                  Events.fire(new InvestmentDelegatedEvent(recommendation, balance.intValue(),
86                          api.getConfirmationProvider().getId()));
87                  if (recommendation.isConfirmationRequired()) {
88                      // confirmation required, delegation successful => forget
89                      tracker.ignoreLoanForever(loanId);
90                  } else {
91                      // confirmation not required, delegation successful => maybe try again in next session
92                      tracker.ignoreLoanForNow(loanId);
93                  }
94                  return Optional.empty();
95              case INVESTED:
96                  final int confirmedAmount = response.getConfirmedAmount().getAsInt();
97                  final Investment i = new Investment(recommendation.getLoanDescriptor().getLoan(), confirmedAmount);
98                  Events.fire(new InvestmentMadeEvent(i, balance.intValue() - confirmedAmount));
99                  tracker.makeInvestment(i);
100                 return Optional.of(i);
101             default:
102                 throw new IllegalStateException("Investment operation failed remotely.");
103         }
104     }

106     /**
107      * Blocked amounts represent loans in various stages. Either the user has invested and the loan has not yet been
108      * funded to 100 % ("na tržišti"), or the user invested and the loan has been funded ("na cestě"). In the latter
109      * case, the loan has already disappeared from the marketplace, which means that it will not be available for
110      * investing any more. As far as I know, the next stage is "v pořádku", the blocked amount is cleared and the loan
111      * becomes an active investment.
112      *
113      * Based on that, this method deals with the first case - when the loan is still available for investing, but we've
114      * already invested as evidenced by the blocked amount. It also unnecessarily deals with the second case, since
115      * that is represented by a blocked amount as well. But that does no harm.
116      *
117      * In case user has made repeated investments into a particular loan, this will show up as multiple blocked amounts.
118      * The method needs to handle this as well.
119      *
120      * @param api Authenticated API that will be used to retrieve the user's blocked amounts from the wallet.
121      * @return Every blocked amount represents a future investment. This method returns such investments.
122      */
123     static List<Investment> retrieveInvestmentsRepresentedByBlockedAmounts(final ZonkyProxy api) {
124         // first group all blocked amounts by the loan ID and sum them
125         final Map<Integer, Integer> amountsBlockedByLoans =
126                 api.execute(zonky -> zonky.getBlockedAmounts(Integer.MAX_VALUE, 0))
127                         .stream()
128                         .filter(blocked -> blocked.getLoanId() > 0) // 0 == Zonky investors' fee
129                         .collect(Collectors.groupingBy(BlockedAmount::getLoanId,
130                                 Collectors.summingInt(BlockedAmount::getAmount)));
131         // and then fetch all the loans in parallel, converting them into investments
132         return Collections.unmodifiableList(amountsBlockedByLoans.entrySet().parallelStream()
133                 .map(entry -> LoanRetriever.getLoan(api, entry.getKey())
134                         .map(l -> new Investment(l, entry.getValue()))
135                         .orElseThrow(() -> new RuntimeException("Loan retrieval failed."))
136                 ).collect(Collectors.toList()));
137     }

139     /**
140      * Zonky API may return {@link ZonkyApi#getStatistics()} as null if the account has no previous investments.
141      *
142      * @param api API to execute the operation.
143      * @return Either what the API returns, or an empty object.
144      */
145     private static Statistics retrieveStatistics(final ZonkyProxy api) {
146         final Statistics returned = api.execute(ZonkyApi::getStatistics);
147         return returned == null ? new Statistics() : returned;
148     }

164     /**
165      * One of the two entry points to the investment API. This takes the strategy, determines suitable loans, and tries
166      * to invest in as many of them as the balance allows.
167      *
168      * @param strategy The investment strategy to use.
169      * @param loans Loans that are available on the marketplace.
170      * @return Investments made in the session.
171      */
172     public Collection<Investment> invest(final InvestmentStrategy strategy, final Collection<LoanDescriptor> loans) {
173         // make sure we have enough money to invest
174         final BigDecimal minimumInvestmentAmount = BigDecimal.valueOf(Defaults.MINIMUM_INVESTMENT_IN_CZK);
175         if (balance.compareTo(minimumInvestmentAmount) < 0) {
176             return Collections.emptyList(); // no need to do anything else
177         }
178         // read our investment statistics
179         final Statistics stats = Investor.retrieveStatistics(this.api);
180         Investor.LOGGER.debug("The sum total of principal remaining on active loans: {} CZK.",
181                 stats.getCurrentOverview().getPrincipalLeft());
182         // figure out which loans we can still put money into
183         final InvestmentTracker tracker = new InvestmentTracker(loans, this.balance);
184         tracker.registerExistingInvestments(Investor.retrieveInvestmentsRepresentedByBlockedAmounts(this.api));
185         // invest the money
186         this.runInvestmentLoop(strategy, tracker, stats, minimumInvestmentAmount);
187         this.balance = tracker.getCurrentBalance();
188         // report
189         final PortfolioOverview portfolio = PortfolioOverview.calculate(balance, stats, tracker.getAllInvestments());
190         Investor.LOGGER.info("Expected annual yield of portfolio: {} % ({} CZK).",
191                 portfolio.getRelativeExpectedYield().scaleByPowerOfTen(2).setScale(2, RoundingMode.HALF_EVEN),
192                 portfolio.getCzkExpectedYield());
193         return tracker.getInvestmentsMade();
194     }
195 
196     private void runInvestmentLoop(final InvestmentStrategy strategy, final InvestmentTracker tracker,
197                                    final Statistics stats, final BigDecimal minimumInvestmentAmount) {
198         Investor.LOGGER.debug("The following available loans have not yet been invested into: {}.",
199                 tracker.getAvailableLoans().stream()
200                         .map(l -> l.getLoan().getId())
201                         .sorted()
202                         .map(String::valueOf)
203                         .collect(Collectors.joining(", ", "[", "]")));
204         Events.fire(new StrategyStartedEvent(strategy, tracker.getAvailableLoans(), balance.intValue()));
205         do {
206             final PortfolioOverview portfolio =
207                     PortfolioOverview.calculate(tracker.getCurrentBalance(), stats, tracker.getAllInvestments());
208             Investor.LOGGER.debug("Current share of unpaid loans with a given rating: {}.",
209                     portfolio.getSharesOnInvestment());
210             final boolean investmentWasMade = strategy.recommend(tracker.getAvailableLoans(), portfolio).stream()
211                     .peek(r -> Events.fire(new LoanRecommendedEvent(r)))
212                     .map(r -> Investor.actuallyInvest(r, this.api, tracker))
213                     .flatMap(o -> o.isPresent() ? Stream.of(o.get()) : Stream.empty())
214                     .findFirst().isPresent();
215             if (!investmentWasMade) { // there is nothing to invest into; RoboZonky is finished now
216                 break;
217             }
218         } while (tracker.getCurrentBalance().compareTo(minimumInvestmentAmount) >= 0);
219         Events.fire(new StrategyCompletedEvent(strategy, tracker.getInvestmentsMade(), balance.intValue()));
220     }

222     /**
223      * One of the two entry points to the investment API. Takes a particular loan and invests a given amount into it.
224      *
225      * @param loanId ID of the loan that should be invested into.
226      * @param loanAmount Amount in CZK to be invested.
227      * @return Present if investment succeeded, empty otherwise.
228      */
229     public Optional<Investment> invest(final int loanId, final int loanAmount, final TemporalAmount captchaDuration) {
230         final Loan l = api.execute(zonky -> zonky.getLoan(loanId));
231         final Optional<Recommendation> r = new LoanDescriptor(l, captchaDuration).recommend(loanAmount, false);
232         final InvestmentTracker t = new InvestmentTracker(Collections.emptyList(), this.balance);
233         final Optional<Investment> result = r.map(r2 -> Investor.actuallyInvest(r2, this.api, t))
234                 .orElse(Optional.empty());
235         this.balance = t.getCurrentBalance();
236         return result;
237     }
238 
239     public BigDecimal getBalance() {
240         return balance;
241     }
