17  package com.github.triceo.robozonky;
18  
19  import java.util.function.Function;
20  
21  import com.github.triceo.robozonky.api.remote.ZonkyApi;
22  import com.github.triceo.robozonky.api.remote.ZonkyOAuthApi;
23  import com.github.triceo.robozonky.api.remote.entities.ZonkyApiToken;
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  
27  /**
28   * Used to authenticate to the Zonky API. Use either {@link #withAccessToken(String, ZonkyApiToken)},
29   * {@link #withAccessTokenAndRefresh(String, ZonkyApiToken)} or
30   * {@link #withCredentials(String, char[])} to log in.
31   */
32  public class Authenticator {
33  
34      private static final Logger LOGGER = LoggerFactory.getLogger(Authenticator.class);
35      private static final String TARGET_SCOPE = "SCOPE_APP_WEB";
36  
37      /**
38       * Prepare for authentication using username and password.
39       *
40       * @param username Zonky username.
41       * @param password Zonky password.
42       * @return Instance ready for authentication.
43       */
44      public static Authenticator withCredentials(final String username, final char[] password) {
45          return new Authenticator(api -> {
46              final ZonkyApiToken token =
47                      api.login(username, new String(password), "password", Authenticator.TARGET_SCOPE);
48              Authenticator.LOGGER.info("Logged in with Zonky as user '{}' using password.", username);
49              return token;
50          }, false);
51      }
52  
53      /**
54       * Prepare for authentication using the Zonky OAuth token.
55       *
56       * @param username Zonky username.
57       * @param token OAuth token.
58       * @return Instance ready for authentication.
59       */
60      public static Authenticator withAccessToken(final String username, final ZonkyApiToken token) {
61          return new Authenticator(api -> {
62              Authenticator.LOGGER.info("Logged in with Zonky as user '{}' with existing access token.", username);
63              return token;
64          }, true);
65      }
66  
67      /**
68       * Prepare for authentication using the Zonky OAuth token which will also refresh the token.
69       *
70       * @param username Zonky username.
71       * @param token OAuth token.
72       * @return Instance ready for authentication.
73       */
74      public static Authenticator withAccessTokenAndRefresh(final String username, final ZonkyApiToken token) {
75          return new Authenticator(api -> {
76              final String tokenId = String.valueOf(token.getRefreshToken());
77              final ZonkyApiToken newToken = api.refresh(tokenId, "refresh_token", Authenticator.TARGET_SCOPE);
78              Authenticator.LOGGER.info("Logged in with Zonky as user '{}', refreshing existing access token.", username);
79              return newToken;
80          }, true);
81      }
82  
83      private final Function<ZonkyOAuthApi, ZonkyApiToken> authenticationMethod;
84      private final boolean tokenBased;
85  
86      private Authenticator(final Function<ZonkyOAuthApi, ZonkyApiToken> authenticationMethod, final boolean tokenBased) {
87          if (authenticationMethod == null) {
88              throw new IllegalArgumentException("Authentication method must be provided.");
89          }
90          this.authenticationMethod = authenticationMethod;
91          this.tokenBased = tokenBased;
92      }
93  
94      /**
95       * Whether or not this particular authentication will use the Zonky OAuth access token.
96       * @return True if token is being used.
97       */
98      public boolean isTokenBased() {
99          return tokenBased;
100     }
101 
102     /**
103      * Perform the actual authentication. Will throw an unchecked exception in case authentication failed.
104      * @param provider The provider to be used when constructing the APIs.
105      * @return Information about the authentication.
106      */
107     public Authentication authenticate(final ApiProvider provider) {
108         final ZonkyApiToken token = authenticationMethod.apply(provider.oauth());
109         return new Authenticator.AuthenticationImpl(provider, token);
110     }
111 
112     private static class AuthenticationImpl implements Authentication {
113 
114         private final ApiProvider provider;
115         private final ZonkyApiToken token;
116 
117         public AuthenticationImpl(final ApiProvider provider, final ZonkyApiToken token) {
118             this.provider = provider;
119             this.token = token;
120         }
121 
122         @Override
123         public ZonkyApi getZonkyApi() {
124             return provider.authenticated(token);
125         }
126 
127         @Override
128         public ZonkyApiToken getZonkyApiToken() {
129             return token;
130         }
131 
132     }
133 }
