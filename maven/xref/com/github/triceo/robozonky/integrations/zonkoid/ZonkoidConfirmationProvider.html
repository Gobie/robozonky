44  /**
45   * This provider will never return {@link ConfirmationType#APPROVED}. If delegated, this provider will allow
46   * investors to fill out CAPTCHA.
47   */
48  public class ZonkoidConfirmationProvider implements ConfirmationProvider {
49  
50      private static final Logger LOGGER = LoggerFactory.getLogger(ZonkoidConfirmationProvider.class);
51      private static final String PROTOCOL_MAIN = "https", PROTOCOL_FALLBACK = "http", CLIENT_APP = "ROBOZONKY";
52      static final String PATH = "/zonkycommander/rest/notifications";
53  
54      static String md5(final String secret) throws NoSuchAlgorithmException {
55          final MessageDigest mdEnc = MessageDigest.getInstance("MD5");
56          mdEnc.update(secret.getBytes(Defaults.CHARSET), 0, secret.length());
57          return new BigInteger(1, mdEnc.digest()).toString(16);
58      }
59  
60      static String getAuthenticationString(final RequestId requestId, final int loanId) {
61          final String auth = new StringJoiner("|")
62                  .add(String.valueOf(requestId.getPassword()))
63                  .add(ZonkoidConfirmationProvider.CLIENT_APP)
64                  .add(requestId.getUserId())
65                  .add(String.valueOf(loanId))
66                  .toString();
67          try {
68              return ZonkoidConfirmationProvider.md5(auth);
69          } catch (final NoSuchAlgorithmException ex) {
70              throw new IllegalStateException("Your Java Runtime Environment does not support MD5!", ex);
71          }
72      }
73  
74      static HttpEntity getFormData(final RequestId requestId, final int loanId, final int amount)
75              throws UnsupportedEncodingException {
76          final List<NameValuePair> nvps = Arrays.asList(
77              new BasicNameValuePair("clientApp", ZonkoidConfirmationProvider.CLIENT_APP),
78              new BasicNameValuePair("username", requestId.getUserId()),
79              new BasicNameValuePair("loanId", String.valueOf(loanId)),
80              new BasicNameValuePair("preferredAmount", String.valueOf(amount))
81          );
82          return new UrlEncodedFormEntity(nvps);
83      }
84  
85      static HttpPost getRequest(final RequestId requestId, final int loanId, final int amount, final String protocol,
86                                 final String domain) throws UnsupportedEncodingException {
87          final String auth = ZonkoidConfirmationProvider.getAuthenticationString(requestId, loanId);
88          final HttpPost httpPost = new HttpPost(protocol + "://" + domain + ZonkoidConfirmationProvider.PATH);
89          httpPost.addHeader("Accept", "text/plain");
90          httpPost.addHeader("Authorization", auth);
91          httpPost.addHeader("Content-Type", "application/x-www-form-urlencoded");
92          httpPost.addHeader("User-Agent", Defaults.ROBOZONKY_USER_AGENT);
93          httpPost.setEntity(ZonkoidConfirmationProvider.getFormData(requestId, loanId, amount));
94          return httpPost;
95      }
96  
97      static Optional<Confirmation> handleError(final RequestId requestId, final int loanId, final int amount,
98                                                final String domain, final String protocol, final Exception ex) {
99          switch (protocol) {
100             case ZonkoidConfirmationProvider.PROTOCOL_MAIN:
101                 ZonkoidConfirmationProvider.LOGGER.warn("HTTPS communication with Zonkoid failed, trying HTTP.");
102                 return ZonkoidConfirmationProvider.requestConfirmation(requestId, loanId, amount, domain,
103                         ZonkoidConfirmationProvider.PROTOCOL_FALLBACK);
104             case ZonkoidConfirmationProvider.PROTOCOL_FALLBACK:
105                 ZonkoidConfirmationProvider.LOGGER.error("Communication with Zonkoid failed.", ex);
106                 return Optional.empty();
107             default:
108                 throw new IllegalStateException("Can not happen.");
109         }
110     }
111 
112     private static Optional<Confirmation> requestConfirmation(final RequestId requestId, final int loanId,
113                                                               final int amount, final String domain,
114                                                               final String protocol) {
115         try (final CloseableHttpClient httpclient = HttpClients.createDefault()) {
116             ZonkoidConfirmationProvider.LOGGER.trace("Sending request.");
117             final HttpPost post = ZonkoidConfirmationProvider.getRequest(requestId, loanId, amount, protocol, domain);
118             try (final CloseableHttpResponse response = httpclient.execute(post)) {
119                 final int statusCode = response.getStatusLine().getStatusCode();
120                 if (statusCode >= 200 && statusCode < 300) {
121                     ZonkoidConfirmationProvider.LOGGER.debug("Response: {}", response.getStatusLine());
122                     return Optional.of(new Confirmation(ConfirmationType.DELEGATED));
123                 } else if (statusCode >= 400 && statusCode < 500){
124                     ZonkoidConfirmationProvider.LOGGER.warn("Response: {}", response.getStatusLine());
125                     return Optional.of(new Confirmation(ConfirmationType.REJECTED));
126                 } else {
127                     ZonkoidConfirmationProvider.LOGGER.error("Unknown response: {}", response.getStatusLine());
128                     return Optional.empty();
129                 }
130             }
131         } catch (final Exception ex) {
132             return ZonkoidConfirmationProvider.handleError(requestId, loanId, amount, domain, protocol, ex);
133         }
134     }
135 
136     static Optional<Confirmation> requestConfirmation(final RequestId requestId, final int loanId, final int amount,
137                                                       final String domain) {
138         return ZonkoidConfirmationProvider.requestConfirmation(requestId, loanId, amount, domain,
139                 ZonkoidConfirmationProvider.PROTOCOL_MAIN);
140     }
141     @Override
142     public Optional<Confirmation> requestConfirmation(final RequestId requestId, final int loanId, final int amount) {
143         return ZonkoidConfirmationProvider.requestConfirmation(requestId, loanId, amount, "urbancoders.eu");
144     }
145 
146     @Override
147     public String getId() {
148         return "Zonkoid / Zonkios";
149     }
