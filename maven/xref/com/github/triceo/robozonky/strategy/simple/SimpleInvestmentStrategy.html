44  class SimpleInvestmentStrategy implements InvestmentStrategy {
45  
46      private static final Logger LOGGER = LoggerFactory.getLogger(SimpleInvestmentStrategy.class);
47      private static final Comparator<LoanDescriptor> BY_TERM =
48              Comparator.comparingInt(l -> l.getLoan().getTermInMonths());
49  
50      static Map<Rating, Collection<LoanDescriptor>> sortLoansByRating(final Collection<LoanDescriptor> loans) {
51          return Collections.unmodifiableMap(loans.stream().distinct()
52                  .collect(Collectors.groupingBy(l -> l.getLoan().getRating())));
53      }
54  
55      private List<Rating> rankRatingsByDemand(final Map<Rating, BigDecimal> currentShare,
56                                               final Function<StrategyPerRating, BigDecimal> metric) {
57          final SortedMap<BigDecimal, List<Rating>> mostWantedRatings = new TreeMap<>(Comparator.reverseOrder());
58          // put the ratings into buckets based on how much we're missing them
59          currentShare.forEach((r, currentRatingShare) -> {
60              final BigDecimal maximumAllowedShare = metric.apply(this.individualStrategies.get(r));
61              final BigDecimal undershare = maximumAllowedShare.subtract(currentRatingShare);
62              if (undershare.compareTo(BigDecimal.ZERO) <= 0) { // we over-invested into this rating; do not include
63                  return;
64              }
65              mostWantedRatings.compute(undershare, (k, v) -> { // each rating unique at source, so list works
66                  final List<Rating> target = (v == null) ? new ArrayList<>(1) : v;
67                  target.add(r);
68                  return target;
69              });
70          });
71          return mostWantedRatings.values().stream().flatMap(Collection::stream).collect(Collectors.toList());
72      }
73  
74      /**
75       * Ranks the ratings in the portfolio which need to have their overall share increased. First on this list will be
76       * ratings which are under their target share. Following them will be ratings which are under their maximum share.
77       * Ratings with their respective shares over the maximum will not be present.
78       *
79       * @param currentShare Current share of investments in a given rating.
80       * @return Ratings in the order of decreasing demand.
81       */
82      List<Rating> rankRatingsByDemand(final Map<Rating, BigDecimal> currentShare) {
83          // find out which ratings are under-invested
84          final List<Rating> ratingsUnderTarget = rankRatingsByDemand(currentShare, StrategyPerRating::getTargetShare);
85          SimpleInvestmentStrategy.LOGGER.info("Ratings under-invested: {}.", ratingsUnderTarget);
86          // find out which other ratings are not yet maxed out
87          final Map<Rating, BigDecimal> filteredShare = currentShare.entrySet()
88                  .stream()
89                  .filter(e -> !ratingsUnderTarget.contains(e.getKey()))
90                  .collect(Collectors.toMap(Map.Entry::getKey, Map.Entry::getValue));
91          final List<Rating> ratingsUnderMaximum = rankRatingsByDemand(filteredShare, StrategyPerRating::getMaximumShare);
92          Collections.reverse(ratingsUnderMaximum); // the closer we get to maximum share, the less desirable
93          SimpleInvestmentStrategy.LOGGER.info("Ratings not yet over-invested: {}.", ratingsUnderMaximum);
94          // merge both and produce result
95          final List<Rating> result = Stream.concat(ratingsUnderTarget.stream(), ratingsUnderMaximum.stream())
96                  .collect(Collectors.toList());
97          return Collections.unmodifiableList(result);
98      }

100     private final int minimumBalance, investmentCeiling;
101     private final Map<Rating, StrategyPerRating> individualStrategies = new EnumMap<>(Rating.class);
102 
103     SimpleInvestmentStrategy(final int minimumBalance, final int investmentCeiling,
104                              final Map<Rating, StrategyPerRating> individualStrategies) {
105         this.minimumBalance = minimumBalance;
106         this.investmentCeiling = investmentCeiling;
107         for (final Rating r: Rating.values()) {
108             if (!individualStrategies.containsKey(r)) {
109                 throw new IllegalArgumentException("Missing strategy for rating " + r);
110             }
111             final StrategyPerRating s = individualStrategies.get(r);
112             this.individualStrategies.put(r, s);
113         }
114     }
115 
116     private boolean isAcceptable(final PortfolioOverview portfolio) {
117         final int availableBalance = portfolio.getCzkAvailable();
118         if (availableBalance < this.minimumBalance) {
119             SimpleInvestmentStrategy.LOGGER.debug("According to the investment strategy, {} CZK balance is less than "
120                     + "minimum {} CZK. Not recommending any loans.", availableBalance, this.minimumBalance);
121             return false;
122         }
123         final int invested = portfolio.getCzkInvested();
124         if (invested > this.investmentCeiling) {
125             SimpleInvestmentStrategy.LOGGER.debug("According to the investment strategy, {} CZK total investment "
126                     + "exceeds {} CZK ceiling. Not recommending any loans.", invested, this.investmentCeiling);
127             return false;
128         }
129         return true;
130     }
131 
132     private boolean needsConfirm(final LoanDescriptor loanDescriptor) {
133         final Rating r = loanDescriptor.getLoan().getRating();
134         return this.individualStrategies.get(r).isConfirmationRequired();
135     }
136 
137     @Override
138     public List<Recommendation> recommend(final Collection<LoanDescriptor> availableLoans,
139                                           final PortfolioOverview portfolio) {
140         if (!this.isAcceptable(portfolio)) {
141             return Collections.emptyList();
142         }
143         final Map<Rating, Collection<LoanDescriptor>> splitByRating =
144                 SimpleInvestmentStrategy.sortLoansByRating(availableLoans);
145         final List<LoanDescriptor> acceptableLoans =
146                 this.rankRatingsByDemand(portfolio.getSharesOnInvestment()).stream()
147                         .filter(splitByRating::containsKey)
148                         .map(this.individualStrategies::get)
149                         .flatMap(strategy -> {
150                             final Comparator<LoanDescriptor> properOrder = strategy.isLongerTermPreferred() ?
151                                     SimpleInvestmentStrategy.BY_TERM.reversed() : SimpleInvestmentStrategy.BY_TERM;
152                             return splitByRating.get(strategy.getRating()).stream()
153                                     .filter(l -> strategy.isAcceptable(l.getLoan()))
154                                     .sorted(properOrder);
155                         }).collect(Collectors.toList());
156         SimpleInvestmentStrategy.LOGGER.debug("Strategy recommends the following unseen loans: {}.", acceptableLoans);
157         return Collections.unmodifiableList(acceptableLoans.stream()
158                 .map(l -> l.recommend(this.recommendInvestmentAmount(l.getLoan(), portfolio), this.needsConfirm(l)))
159                 .filter(Optional::isPresent)
160                 .map(Optional::get)
161                 .collect(Collectors.toList()));
162     }
163 
164     int recommendInvestmentAmount(final Loan loan, final PortfolioOverview portfolio) {
165         if (!this.isAcceptable(portfolio)) {
166             return 0;
167         }
168         final Optional<int[]> recommend = individualStrategies.get(loan.getRating()).recommendInvestmentAmount(loan);
169         if (!recommend.isPresent()) { // does not match the strategy
170             return 0;
171         }
172         final int minimumRecommendation = recommend.get()[0];
173         final int maximumRecommendation = recommend.get()[1];
174         SimpleInvestmentStrategy.LOGGER.trace("Recommended investment for loan #{} in range of <{}; {}> CZK.",
175                 loan.getId(), minimumRecommendation, maximumRecommendation);
176         // round to nearest lower increment
177         final int balance = portfolio.getCzkAvailable();
178         final int loanRemaining = (int)loan.getRemainingInvestment();
179         if (minimumRecommendation > balance) {
180             return 0;
181         } else if (minimumRecommendation > loanRemaining) {
182             return 0;
183         }
184         final int maxAllowedInvestmentIncrement = Defaults.MINIMUM_INVESTMENT_INCREMENT_IN_CZK;
185         final int recommendedAmount = Math.min(balance, Math.min(maximumRecommendation, loanRemaining));
186         final int result = (recommendedAmount / maxAllowedInvestmentIncrement) * maxAllowedInvestmentIncrement;
187         if (result < minimumRecommendation) {
188             return 0;
189         } else {
190             SimpleInvestmentStrategy.LOGGER.debug("Final recommendation for loan #{} is {} CZK.", loan.getId(), result);
191             return result;
192         }
193     }
