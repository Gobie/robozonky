40          return ImmutableConfiguration.from(strategy);

43      static int getMinimumBalance(final ImmutableConfiguration config) {
44          final int minimumBalance = StrategyFileProperty.MINIMUM_BALANCE.getValue(config::getInt)
45                  .orElseThrow(() -> new IllegalStateException("Minimum balance is missing."));
46          if (minimumBalance < Defaults.MINIMUM_INVESTMENT_IN_CZK) {
47              throw new IllegalStateException("Minimum balance < " + Defaults.MINIMUM_INVESTMENT_IN_CZK + " CZK.");
48          }
49          return minimumBalance;
50      }
51  
52      static int getMaximumInvestment(final ImmutableConfiguration config) {
53          final int maximumInvestment = StrategyFileProperty.MAXIMUM_INVESTMENT.getValue(config::getInt)
54                  .orElseThrow(() -> new IllegalStateException("Maximum investment is missing."));
55          if (maximumInvestment < Defaults.MINIMUM_INVESTMENT_IN_CZK) {
56              throw new IllegalStateException("Maximum investment < " + Defaults.MINIMUM_INVESTMENT_IN_CZK + " CZK.");
57          }
58          return maximumInvestment;
59      }

61      private static StrategyPerRating createIndividualStrategy(final Rating r, final BigDecimal targetShare,
62                                                                final BigDecimal maxShare, final int minTerm,
63                                                                final int maxTerm, final int minAskAmount,
64                                                                final int maxAskAmount, final int minLoanAmount,
65                                                                final int maxLoanAmount, final BigDecimal minLoanShare,
66                                                                final BigDecimal maxLoanShare,
67                                                                final boolean preferLongerTerms,
68                                                                final boolean requireConfirmation) {
69          if (Objects.equals(targetShare, BigDecimal.ZERO) && Objects.equals(maxShare, BigDecimal.ZERO)) {
70              SimpleInvestmentStrategyService.LOGGER.debug("Rating '{}' will never be invested into.", r.getCode());
71          } else {
72              SimpleInvestmentStrategyService.LOGGER.debug("Acceptable shares of rating '{}' on total investments is " +
73                      "between {} and {}.", r.getCode(), targetShare, maxShare);
74              SimpleInvestmentStrategyService.LOGGER.debug("Acceptable loans for rating '{}' are between {} and {} CZK on " +
75                              "terms between {} and {} months, {} preferred.", r.getCode(), minAskAmount,
76                      maxAskAmount < 0 ? "+inf" : maxAskAmount, minTerm == -1 ? 0 : minTerm,
77                      maxTerm < 0 ? "+inf" : maxTerm + 1, preferLongerTerms ? "longer" : "shorter");
78              SimpleInvestmentStrategyService.LOGGER.debug("Acceptable investment for a loan of rating '{}' is between {} " +
79                              "and {} CZK with investment share between {} and {}.", r.getCode(), minLoanAmount,
80                      maxLoanAmount < 0 ? "+inf" : maxLoanAmount, minLoanShare, maxLoanShare);
81              if (requireConfirmation) {
82                  SimpleInvestmentStrategyService.LOGGER.debug("Rating '{}' requires confirmation.", r.getCode());
83              }
84          }
85          return new StrategyPerRating(r, targetShare, maxShare, minTerm, maxTerm, minLoanAmount, maxLoanAmount,
86                  minLoanShare, maxLoanShare, minAskAmount, maxAskAmount, preferLongerTerms, requireConfirmation);
87      }

89      private static StrategyPerRating parseRating(final Rating rating, final ImmutableConfiguration config) {
90          SimpleInvestmentStrategyService.LOGGER.debug("- Adding strategy for rating '{}'.", rating.getCode());
91          final boolean requireApproval = StrategyFileProperty.REQUIRE_CONFIRMATION.getValue(rating, config::getBoolean);
92          final boolean preferLongerTerms = StrategyFileProperty.PREFER_LONGER_TERMS.getValue(rating, config::getBoolean);
93          // shares of a rating on the total pie
94          final BigDecimal targetShare = StrategyFileProperty.TARGET_SHARE.getValue(rating, config::getBigDecimal);
95          if (!Util.isBetweenZeroAndOne(targetShare)) {
96              throw new IllegalStateException("Target share for rating " + rating + " outside of range <0, 1>: "
97                      + targetShare);
98          }
99          final BigDecimal maximumShare = StrategyFileProperty.MAXIMUM_SHARE.getValue(rating, config::getBigDecimal);
100         if (!Util.isBetweenAAndB(maximumShare, targetShare, BigDecimal.ONE)) {
101             throw new IllegalStateException("Maximum share for rating " + rating + " outside of range <" + targetShare
102                     + ", 1>: " + maximumShare);
103         }
104         // terms
105         final int minTerm = StrategyFileProperty.MINIMUM_TERM.getValue(rating, config::getInt);
106         if (minTerm < -1) {
107             throw new IllegalStateException("Minimum acceptable term for rating " + rating + " negative.");
108         }
109         final int maxTerm = StrategyFileProperty.MAXIMUM_TERM.getValue(rating, config::getInt);
110         if (maxTerm < -1) {
111             throw new IllegalStateException("Maximum acceptable term for rating " + rating + " negative.");
112         } else if (minTerm > maxTerm && maxTerm != -1) {
113             throw new IllegalStateException("Maximum acceptable term for rating " + rating + " less than minimum.");
114         }
115         // loan asks
116         final int minAskAmount = StrategyFileProperty.MINIMUM_ASK.getValue(rating, config::getInt);
117         if (minAskAmount < -1) {
118             throw new IllegalStateException("Minimum acceptable ask amount for rating " + rating + " negative.");
119         }
120         final int maxAskAmount = StrategyFileProperty.MAXIMUM_ASK.getValue(rating, config::getInt);
121         if (maxAskAmount < -1) {
122             throw new IllegalStateException("Maximum acceptable ask for rating " + rating + " negative.");
123         } else if (minAskAmount > maxAskAmount && maxAskAmount != -1) {
124             throw new IllegalStateException("Maximum acceptable ask for rating " + rating + " less than minimum.");
125         }
126         // investment amounts
127         final int minLoanAmount = StrategyFileProperty.MINIMUM_LOAN_AMOUNT.getValue(rating, config::getInt);
128         if (minLoanAmount < Defaults.MINIMUM_INVESTMENT_IN_CZK) {
129             throw new IllegalStateException("Minimum investment amount for rating " + rating
130                     + " less than minimum investment allowed: " + minLoanAmount + " CZK.");
131         }
132         final int maxLoanAmount = StrategyFileProperty.MAXIMUM_LOAN_AMOUNT.getValue(rating, config::getInt);
133         if (maxLoanAmount != -1 && maxLoanAmount < minLoanAmount) {
134             throw new IllegalStateException("Maximum investment amount for rating " + rating + " less than minimum.");
135         }
136         final BigDecimal minLoanShare = StrategyFileProperty.MINIMUM_LOAN_SHARE.getValue(rating,
137                 config::getBigDecimal);
138         if (!Util.isBetweenZeroAndOne(minLoanShare)) {
139             throw new IllegalStateException("Minimum investment share for rating " + rating
140                     + " outside of range <0, 1>: " + targetShare);
141         }
142         final BigDecimal maxLoanShare = StrategyFileProperty.MAXIMUM_LOAN_SHARE.getValue(rating, config::getBigDecimal);
143         if (!Util.isBetweenAAndB(maxLoanShare, minLoanShare, BigDecimal.ONE)) {
144             throw new IllegalStateException("Maximum investment share for rating " + rating
145                     + " outside of range (min, 1>: " + targetShare);
146         }
147         return SimpleInvestmentStrategyService.createIndividualStrategy(rating, targetShare, maximumShare,
148                 minTerm, maxTerm, minAskAmount, maxAskAmount, minLoanAmount, maxLoanAmount, minLoanShare,
149                 maxLoanShare, preferLongerTerms, requireApproval);
150     }

152     private static void checkIndividualStrategies(final Map<Rating, StrategyPerRating> strategies) {
153         final BigDecimal ratingShareSum = strategies.values().stream()
154                 .map(StrategyPerRating::getTargetShare)
155                 .reduce(BigDecimal.ZERO, BigDecimal::add);
156         if (ratingShareSum.compareTo(BigDecimal.ONE) > 0) {
157             throw new IllegalStateException("Sum total of target shares is larger than 1.00: " + ratingShareSum + ".");
158         }
159     }

161     @Override
162     public Optional<InvestmentStrategy> parse(final InputStream strategy)  {
163         try {
164             final ImmutableConfiguration c = SimpleInvestmentStrategyService.getConfig(strategy);
165             final int minimumBalance = SimpleInvestmentStrategyService.getMinimumBalance(c);
166             SimpleInvestmentStrategyService.LOGGER.debug("Minimum balance to invest must be {} CZK.", minimumBalance);
167             final int maximumInvestment = SimpleInvestmentStrategyService.getMaximumInvestment(c);
168             SimpleInvestmentStrategyService.LOGGER.debug("Maximum investment must not exceed {} CZK.", maximumInvestment);
169             final Map<Rating, StrategyPerRating> individualStrategies = Arrays.stream(Rating.values())
170                     .collect(Collectors.toMap(Function.identity(), r -> SimpleInvestmentStrategyService.parseRating(r, c)));
171             SimpleInvestmentStrategyService.checkIndividualStrategies(individualStrategies);
172             return Optional.of(new SimpleInvestmentStrategy(minimumBalance, maximumInvestment, individualStrategies));
173         } catch (final RuntimeException ex) {
174             SimpleInvestmentStrategyService.LOGGER.info("Failed parsing strategy.", ex);
175             return Optional.empty();
176         }
177     }
