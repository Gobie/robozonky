17  package com.github.triceo.robozonky.app;
18  
19  import java.io.File;
20  import java.io.IOException;
21  import java.io.RandomAccessFile;
22  import java.nio.channels.FileLock;
23  import java.util.Optional;
24  import java.util.function.Consumer;
25  
26  import com.github.triceo.robozonky.api.ReturnCode;
27  import org.slf4j.Logger;
28  import org.slf4j.LoggerFactory;
29  
30  /**
31   * Used to ensure that only one RoboZonky process may be executing at any one time. See {@link #get()} for the
32   * contract. Will not mutually exclude RoboZonky instances running within the same JVM.
33   */
34  final class Exclusivity implements ShutdownHook.Handler {
35  
36      private static final Logger LOGGER = LoggerFactory.getLogger(Exclusivity.class);
37  
38      private static final class LockControl {
39  
40          private final RandomAccessFile randomAccessFile;
41          private final FileLock lock;
42  
43          public LockControl(final File lockedFile) throws IOException {
44              this.randomAccessFile = new RandomAccessFile(lockedFile, "rw");
45              this.lock = this.randomAccessFile.getChannel().lock();
46          }
47  
48          public boolean hasValidLock() {
49              return lock.isValid();
50          }
51  
52          public void invalidateLock() throws IOException {
53              this.randomAccessFile.close();
54          }
55  
56      }
57  
58      private final File fileToLock;
59      private Exclusivity.LockControl lockControl = null;
60  
61      /**
62       * Guarantees exclusivity based on a randomly created temp file. This is only useful for testing, since in all
63       * other cases multiple JVMs need to synchronize around the same file.
64       * @throws IOException When the temp file could not be created.
65       */
66      Exclusivity() throws IOException {
67          this(File.createTempFile("robozonky-", ".lock"));
68      }
69  
70      /**
71       * Guarantees exclusivity around a provided file.
72       * @param lock File to lock on.
73       */
74      public Exclusivity(final File lock) {
75          this.fileToLock = lock;
76      }
77  
78      File getFileToLock() {
79          return this.fileToLock;
80      }
81  
82      /**
83       * Whether or not the application has managed to acquire exclusivity.
84       * @return  True between any two successful successive calls to {@link #ensure()} and {@link #waive()} respectively.
85       * Will be false before first successful {@link #ensure()} or after every {@link #waive()}.
86       */
87      synchronized boolean isEnsured() {
88          return lockControl != null && lockControl.hasValidLock();
89      }
90  
91      /**
92       * Allow other instances of RoboZonky to run, possibly acquiring exclusivity for themselves.
93       */
94      synchronized void waive() {
95          if (!this.isEnsured()) {
96              Exclusivity.LOGGER.debug("Already waived.");
97              return;
98          }
99          try {
100             this.lockControl.invalidateLock();
101             Exclusivity.LOGGER.debug("File lock released.");
102         } catch (final IOException ex) {
103             Exclusivity.LOGGER.warn("Failed releasing lock, new RoboZonky processes may not launch.", ex);
104         } finally {
105             Exclusivity.LOGGER.debug("Lock file deleted successfully: {}.", this.fileToLock.delete());
106         }
107     }
108 
109     /**
110      * Will return after this instance of RoboZonky has acquired exclusivity, or failed. May block for extensive periods
111      * of time. Make sure to call {@link #isEnsured()} afterwards to see if exclusivity is ensured.
112      */
113     synchronized void ensure() {
114         if (this.isEnsured()) {
115             Exclusivity.LOGGER.debug("Already ensured.");
116             return;
117         }
118         Exclusivity.LOGGER.info("Checking we're the only RoboZonky running.");
119         Exclusivity.LOGGER.debug("Acquiring file lock: {}.", this.fileToLock.getAbsolutePath());
120         try {
121             this.lockControl = new Exclusivity.LockControl(this.fileToLock);
122             Exclusivity.LOGGER.debug("File lock acquired.");
123         } catch (final IOException ex) {
124             Exclusivity.LOGGER.error("Failed acquiring lock, another RoboZonky process likely running.", ex);
125         }
126     }
127 
128     /**
129      * Call in order to acquire exclusivity. May block for extensive periods of time until exclusivity is acquired.
130      * @return If present, call in order to waive exclusivity. If empty, exclusivity not acquired.
131      */
132     @Override
133     public Optional<Consumer<ReturnCode>> get() {
134         this.ensure();
135         if (!this.isEnsured()) {
136             return Optional.empty();
137         } else { // other RoboZonky instances can now start executing
138             return Optional.of((code) -> this.waive());
139         }
140     }
141 }
