16  package com.github.triceo.robozonky.api.remote;
17  
18  import java.util.Collection;
19  import java.util.List;
20  import javax.ws.rs.Consumes;
21  import javax.ws.rs.DefaultValue;
22  import javax.ws.rs.GET;
23  import javax.ws.rs.HeaderParam;
24  import javax.ws.rs.POST;
25  import javax.ws.rs.Path;
26  import javax.ws.rs.PathParam;
27  import javax.ws.rs.Produces;
28  import javax.ws.rs.QueryParam;
29  import javax.ws.rs.core.MediaType;
30  
31  import com.github.triceo.robozonky.api.remote.entities.BlockedAmount;
32  import com.github.triceo.robozonky.api.remote.entities.Investment;
33  import com.github.triceo.robozonky.api.remote.entities.Loan;
34  import com.github.triceo.robozonky.api.remote.entities.Statistics;
35  import com.github.triceo.robozonky.api.remote.entities.Wallet;
36  import com.github.triceo.robozonky.api.remote.enums.InvestmentStatuses;
37  import com.github.triceo.robozonky.api.remote.enums.Ratings;
38  import org.slf4j.Logger;
39  import org.slf4j.LoggerFactory;
40  
41  @Path("/")
42  @Produces(MediaType.APPLICATION_JSON + ";charset=UTF-8")
43  @Consumes(MediaType.APPLICATION_JSON + ";charset=UTF-8")
44  public interface ZonkyApi extends Api {
45  
46      Logger LOGGER = LoggerFactory.getLogger(ZonkyApi.class);
47  
48      String LOANS = "/loans";
49      String MARKETPLACE = LOANS + "/marketplace";
50      String ME = "/users/me";
51      String WALLET = ME + "/wallet";
52      String INVESTMENT = ME + "/investment";
53      String INVESTMENTS = ME + "/investments";
54  
55      @GET
56      @Path(ZonkyApi.WALLET)
57      Wallet getWallet();
58  
59      /**
60       * Retrieve blocked amounts in the wallet, that either represent a pending investment, or investor's fee. This
61       * query is paginated.
62       *
63       * @param pageSize How many items should be listed on the page.
64       * @param pageNo Number of the page to show, where 0 is the first page.
65       * @return Blocked amount on that particular page.
66       */
67      @GET
68      @Path(ZonkyApi.WALLET + "/blocked-amounts")
69      List<BlockedAmount> getBlockedAmounts(@HeaderParam("X-Size") int pageSize, @HeaderParam("X-Page") int pageNo);
70  
71      @GET
72      @Path(ZonkyApi.LOANS + "/{loanId}")
73      Loan getLoan(@PathParam("loanId") int loanId);
74  
75      @GET
76      @Path(ZonkyApi.MARKETPLACE)
77      List<Loan> getLoans(
78              @QueryParam("rating__in") Ratings ratings,
79              @QueryParam("remainingInvestment__gt") @DefaultValue("0") int leastRemainingInvestment,
80              @QueryParam("termInMonths__gte") @DefaultValue("0") int leastPossibleTermInMonths,
81              @QueryParam("termInMonths__lte") int mostPossibleTermInMonths);
82  
83      @GET
84      @Path(ZonkyApi.MARKETPLACE)
85      List<Loan> getLoans(
86              @QueryParam("rating__in") Ratings ratings,
87              @QueryParam("remainingInvestment__gt") @DefaultValue("0") int leastRemainingInvestment,
88              @QueryParam("termInMonths__gte") @DefaultValue("0") int leastPossibleTermInMonths);
89  
90      @GET
91      @Path(ZonkyApi.MARKETPLACE)
92      List<Loan> getLoans(
93              @QueryParam("rating__in") Ratings ratings,
94              @QueryParam("remainingInvestment__gt") @DefaultValue("0") int leastRemainingInvestment);
95  
96      @GET
97      @Path(ZonkyApi.MARKETPLACE)
98      List<Loan> getLoans(@QueryParam("rating__in") Ratings ratings);
99  
100     @GET
101     @Path(ZonkyApi.MARKETPLACE)
102     @Override
103     List<Loan> getLoans();
104 
105 
106     @GET
107     @Path(ZonkyApi.ME + "/logout")
108     void logout();
109 
110     @GET
111     @Path(ZonkyApi.INVESTMENTS + "/statistics")
112     Statistics getStatistics();
113 
114     @GET
115     @Path(ZonkyApi.INVESTMENTS)
116     Collection<Investment> getInvestments(
117             @QueryParam("loan.status__in") InvestmentStatuses statuses,
118             @QueryParam("loan.dpd__gt") @DefaultValue("0") int daysPastDueGreaterThan);
119 
120     @GET
121     @Path(ZonkyApi.INVESTMENTS)
122     Collection<Investment> getInvestments(@QueryParam("loan.status__in") InvestmentStatuses statuses);
123 
124     @POST
125     @Path("/marketplace/investment")
126     void invest(Investment investment);
127 }
